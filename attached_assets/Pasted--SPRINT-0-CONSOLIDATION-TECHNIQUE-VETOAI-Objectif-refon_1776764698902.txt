Crée le fichier server/lib/response.ts :

  export const ok = (data: any, meta?: Record<string, any>) =>
    ({ success: true, data, ...meta });

  export const fail = (code: string, message: string, details?: any) =>
    ({ success: false, error: { code, message, ...(details && { details }) } });

Mets à jour TOUTES les routes Express pour utiliser ce format :
- Chaque res.json([...]) devient res.json(ok([...]))
- Chaque res.json({...}) avec données devient res.json(ok({...}))
- Pour les listes : res.json(ok(items, { total: items.length, page: 1, pages: 1 }))
- Chaque res.status(400).json({ error: "..." }) devient
  res.status(400).json(fail('CODE_ERREUR', 'message lisible'))

Codes d'erreur à utiliser :
  VALIDATION_ERROR, NOT_FOUND, UNAUTHORIZED, FORBIDDEN,
  PATIENT_CREATE_FAILED, CONSULTATION_SAVE_FAILED,
  ACTE_SAVE_FAILED, FACTURE_CREATE_FAILED, INTERNAL_ERROR

Crée server/lib/validate.ts :

  import { ZodSchema } from 'zod';
  export const validate = (schema: ZodSchema) =>
    (req: any, res: any, next: any) => {
      const result = schema.safeParse(req.body);
      if (!result.success) {
        const details = result.error.errors.map(e => ({
          field: e.path.join('.'),
          message: e.message
        }));
        return res.status(422).json(
          fail('VALIDATION_ERROR', 'Données invalides', details)
        );
      }
      req.body = result.data;
      next();
    };

  import rateLimit from 'express-rate-limit';

    contentSecurityPolicy: {
      directives: {
        defaultSrc: ["'self'"],
        scriptSrc: ["'self'", "'unsafe-inline'", "'unsafe-eval'",
                    "https://clerk.com", "https://*.clerk.accounts.dev"],
        connectSrc: ["'self'", "https://api.clerk.com",
                     "https://*.clerk.accounts.dev", "https://api.anthropic.com"],
        imgSrc: ["'self'", "data:", "https:", "blob:"],
        frameSrc: ["'self'", "https://clerk.com"],
        fontSrc: ["'self'", "https:", "data:"],
      }
    },
    crossOriginEmbedderPolicy: false,
  }));

  // Rate limiting global : 200 req/min par IP
  app.use('/api/', rateLimit({
    windowMs: 60 * 1000,
    max: 200,
    standardHeaders: true,
    legacyHeaders: false,
    message: fail('RATE_LIMIT', 'Trop de requêtes, réessayez dans une minute'),
  }));

  // Rate limiting strict sur les appels IA (coûteux)
  const aiLimiter = rateLimit({ windowMs: 60000, max: 15,
    message: fail('AI_RATE_LIMIT', 'Limite IA atteinte, attendez 1 minute') });
  app.use('/api/consultations/*/diagnostic', aiLimiter);
  app.use('/api/certificats/*/generate', aiLimiter);
  app.use('/api/ordonnances/*/generate', aiLimiter);

Crée client/src/lib/apiClient.ts :

  type ApiResponse<T> = { success: true; data: T; total?: number; pages?: number }
                      | { success: false; error: { code: string; message: string } };

  async function request<T>(url: string, options?: RequestInit): Promise<T> {
    const res = await fetch(url, {
      ...options,
      headers: { 'Content-Type': 'application/json', ...options?.headers },
    });
    const json: ApiResponse<T> = await res.json();
    if (!json.success) {
      throw new Error(json.error.message);
    }
    return json.data;
  }

  export const api = {
    get:    <T>(url: string) => request<T>(url),
    post:   <T>(url: string, body: any) =>
              request<T>(url, { method: 'POST', body: JSON.stringify(body) }),
    put:    <T>(url: string, body: any) =>
              request<T>(url, { method: 'PUT', body: JSON.stringify(body) }),
    delete: <T>(url: string) =>
              request<T>(url, { method: 'DELETE' }),
  };
