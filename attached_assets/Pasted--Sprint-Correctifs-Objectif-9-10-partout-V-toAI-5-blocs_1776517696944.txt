**2C. Si le modal AjouterActeModal n'existe pas — le créer :**

```typescript
// client/src/components/AjouterActeModal.tsx
import { useState } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { apiRequest } from '@/lib/queryClient';
import { useToast } from '@/hooks/use-toast';

interface AjouterActeModalProps {
  consultationId: number;
  patientId: number;
  onClose: () => void;
  onActeAdded: (acte: any) => void;
}

export function AjouterActeModal({ consultationId, patientId, onClose, onActeAdded }: AjouterActeModalProps) {
  const { toast } = useToast();
  const [description, setDescription] = useState('');
  const [quantite, setQuantite] = useState(1);
  const [prixUnitaireHt, setPrixUnitaireHt] = useState('');
  const [tvaRate, setTvaRate] = useState(20);

  const mutation = useMutation({
    mutationFn: async () => {
      const res = await apiRequest('POST', `/api/consultations/${consultationId}/actes`, {
        description,
        quantite,
        prixUnitaireHt: parseFloat(prixUnitaireHt),
        tvaRate,
      });
      return res.json();
    },
    onSuccess: (data) => {
      toast({ title: 'Acte ajouté avec succès' });
      onActeAdded(data);
    },
    onError: () => {
      toast({ title: 'Erreur lors de l\'ajout', variant: 'destructive' });
    },
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (!description.trim() || !prixUnitaireHt) {
      toast({ title: 'Description et prix obligatoires', variant: 'destructive' });
      return;
    }
    mutation.mutate();
  };

  return (
    <Dialog open onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Ajouter un acte / médicament</DialogTitle>
        </DialogHeader>
        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label>Description *</label>
            <Input
              value={description}
              onChange={e => setDescription(e.target.value)}
              placeholder="Ex: Consultation, Radiographie, Meloxicam 1mg/mL..."
            />
          </div>
          <div className="grid grid-cols-2 gap-4">
            <div>
              <label>Quantité</label>
              <Input
                type="number"
                min={1}
                value={quantite}
                onChange={e => setQuantite(Number(e.target.value))}
              />
            </div>
            <div>
              <label>Prix unitaire HT (€) *</label>
              <Input
                type="number"
                step="0.01"
                min="0"
                value={prixUnitaireHt}
                onChange={e => setPrixUnitaireHt(e.target.value)}
                placeholder="Ex: 35.00"
              />
            </div>
          </div>
          <div>
            <label>TVA (%)</label>
            <select
              value={tvaRate}
              onChange={e => setTvaRate(Number(e.target.value))}
              className="w-full border rounded p-2"
            >
              <option value={20}>20% (standard)</option>
              <option value={10}>10% (réduit)</option>
              <option value={5.5}>5.5%</option>
              <option value={0}>0%</option>
            </select>
          </div>
          <div className="flex gap-2 justify-end">
            <Button type="button" variant="outline" onClick={onClose}>Annuler</Button>
            <Button type="submit" disabled={mutation.isPending}>
              {mutation.isPending ? 'Ajout...' : 'Ajouter l\'acte'}
            </Button>
          </div>
        </form>
      </DialogContent>
    </Dialog>
  );
}
```

**2D. Route serveur si elle n'existe pas :**

```typescript
// server/routes/consultations.ts
// Ajouter un acte à une consultation existante
router.post('/consultations/:id/actes', authenticate, async (req, res) => {
  const consultationId = parseInt(req.params.id);
  const { description, quantite, prixUnitaireHt, tvaRate } = req.body;
  
  if (!description || !prixUnitaireHt) {
    return res.status(400).json({ error: 'Description et prix obligatoires' });
  }
  
  const montantHt = quantite * parseFloat(prixUnitaireHt);
  const montantTva = montantHt * (tvaRate / 100);
  const montantTtc = montantHt + montantTva;
  
  const [acte] = await db.insert(lignesFacture).values({
    consultationId,
    description,
    quantite,
    prixUnitaireHt: prixUnitaireHt.toString(),
    tvaRate: tvaRate.toString(),
    totalHt: montantHt.toString(),
  }).returning();
  
  // Recalculer le total de la facture associée
  const facture = await db.select()
    .from(factures)
    .where(eq(factures.consultationId, consultationId))
    .get();
    
  if (facture) {
    const allLignes = await db.select()
      .from(lignesFacture)
      .where(eq(lignesFacture.consultationId, consultationId));
    
    const totalHt = allLignes.reduce((sum, l) => sum + parseFloat(l.totalHt || '0'), 0);
    const totalTva = allLignes.reduce((sum, l) => 
      sum + (parseFloat(l.totalHt || '0') * (parseFloat(l.tvaRate || '20') / 100)), 0);
    const totalTtc = totalHt + totalTva;
    
    await db.update(factures)
      .set({ montantHt: totalHt.toString(), montantTtc: totalTtc.toString() })
      .where(eq(factures.id, facture.id));
  }
  
  res.json(acte);
});
```

router.post('/factures', authenticate, async (req, res) => {
  const { consultationId, patientId } = req.body;
  
  // Récupérer les actes de la consultation
  const lignes = await db.select()
    .from(lignesFacture)
    .where(eq(lignesFacture.consultationId, consultationId));

      error: 'Impossible de créer une facture sans actes enregistrés.'

  const totalHt = lignes.reduce((sum, l) => sum + parseFloat(l.totalHt || '0'), 0);

    sum + (parseFloat(l.totalHt || '0') * (parseFloat(l.tvaRate || '20') / 100)), 0);
  const totalTtc = totalHt + totalTva;
  
  // Générer le numéro de facture
  const year = new Date().getFullYear();
  const count = await db.select({ count: sql`COUNT(*)` })
    .from(factures)
    .where(and(
      eq(factures.clinicId, req.auth.clinicId),
      sql`EXTRACT(YEAR FROM created_at) = ${year}`
    ))
    .get();

    numero,
    statut: 'en_attente',
    montantHt: totalHt.toString(),
    montantTtc: totalTtc.toString(),
    dateEmission: new Date().toISOString().split('T')[0],

  res.json(facture);

## BLOC 4 — FORMATAGE DES DATES (PARTOUT)

### Créer un helper partagé

```typescript
// client/src/lib/formatters.ts

export const formatDate = (dateStr: string | Date | null | undefined): string => {
  if (!dateStr) return '—';
  const d = new Date(dateStr);
  if (isNaN(d.getTime())) return String(dateStr);
  return d.toLocaleDateString('fr-FR', {
    day: 'numeric',
    month: 'long',
    year: 'numeric',
    timeZone: 'Europe/Paris',
  });
};

export const formatDateShort = (dateStr: string | Date | null | undefined): string => {
  if (!dateStr) return '—';
  const d = new Date(dateStr);
  if (isNaN(d.getTime())) return String(dateStr);
  return d.toLocaleDateString('fr-FR', {
    day: '2-digit',
    month: '2-digit',
    year: 'numeric',
    timeZone: 'Europe/Paris',
  });
};

export const formatDateTime = (dateStr: string | Date | null | undefined): string => {
  if (!dateStr) return '—';
  const d = new Date(dateStr);
  if (isNaN(d.getTime())) return String(dateStr);
  return d.toLocaleString('fr-FR', {
    day: '2-digit',
    month: '2-digit',
    year: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
    timeZone: 'Europe/Paris',
  });
};

export const formatCurrency = (amount: number | string | null | undefined): string => {
  const n = Number(amount || 0);
  return new Intl.NumberFormat('fr-FR', { style: 'currency', currency: 'EUR' }).format(n);
};
```
