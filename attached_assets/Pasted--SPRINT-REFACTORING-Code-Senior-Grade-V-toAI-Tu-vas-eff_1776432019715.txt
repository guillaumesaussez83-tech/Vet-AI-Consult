```typescript
// Middleware de validation générique
export function validate<T>(schema: z.ZodSchema<T>) {
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) {
      return res.status(400).json({
        success: false,
        error: 'Données invalides',
        details: result.error.flatten().fieldErrors,
      });
    }
    req.body = result.data; // données nettoyées et typées
    next();
  };
}
```

## BLOC 3 — ERROR HANDLING ROBUSTE

### 3.1 — Créer `server/middleware/errorHandler.ts`
```typescript
import { Request, Response, NextFunction } from 'express';

export class AppError extends Error {
  constructor(
    public statusCode: number,
    public message: string,
    public code?: string,
    public details?: unknown
  ) {
    super(message);
    this.name = 'AppError';
  }
}

export class NotFoundError extends AppError {
  constructor(resource: string) {
    super(404, `${resource} introuvable`, 'NOT_FOUND');
  }
}

export class UnauthorizedError extends AppError {
  constructor(message = 'Non autorisé') {
    super(401, message, 'UNAUTHORIZED');
  }
}

export class ValidationError extends AppError {
  constructor(message: string, details?: unknown) {
    super(400, message, 'VALIDATION_ERROR', details);
  }
}

// Middleware global d'erreur Express
export function errorHandler(
  err: Error,
  req: Request,
  res: Response,
  next: NextFunction
): void {
  if (err instanceof AppError) {
    res.status(err.statusCode).json({
      success: false,
      error: err.message,
      code: err.code,
      details: err.details,
    });
    return;
  }

  // Erreur Zod
  if (err.name === 'ZodError') {
    res.status(400).json({
      success: false,
      error: 'Données invalides',
      code: 'VALIDATION_ERROR',
    });
    return;
  }

  // Erreur PostgreSQL
  if ((err as any).code === '23505') {
    res.status(409).json({
      success: false,
      error: 'Cette ressource existe déjà',
      code: 'DUPLICATE_ERROR',
    });
    return;
  }

  // Erreur inconnue — ne pas exposer les détails en production
  console.error('[ERROR]', err);
  res.status(500).json({
    success: false,
    error: process.env.NODE_ENV === 'production' 
      ? 'Erreur serveur interne' 
      : err.message,
    code: 'INTERNAL_ERROR',
  });
}
```

```typescript
type AsyncHandler = (req: Request, res: Response, next: NextFunction) => Promise<void>;

export const asyncHandler = (fn: AsyncHandler) =>

    Promise.resolve(fn(req, res, next)).catch(next);
```

## BLOC 5 — ARCHITECTURE PROPRE

### 5.1 — Séparer les services des routes
Créer un dossier `server/services/` avec un fichier par domaine métier :

**`server/services/factureService.ts`**
```typescript
import { db } from '../db';
import { factures, lignesFacture } from '../schema';
import { eq, and } from 'drizzle-orm';
import { AppError } from '../middleware/errorHandler';
import type { CreateFactureInput } from '../../shared/types';

export class FactureService {
  // Recalcul toujours depuis les lignes — jamais depuis le frontend
  static async recalculerMontants(factureId: number): Promise<{
    montantHt: number;
    montantTva: number;
    montantTtc: number;
  }> {
    const lignes = await db.select()
      .from(lignesFacture)
      .where(eq(lignesFacture.factureId, factureId));

    const montantHt = lignes.reduce((sum, l) => 
      sum + (Number(l.quantite) * Number(l.prixUnitaireHt)), 0);
    
    const montantTva = lignes.reduce((sum, l) => 
      sum + (Number(l.quantite) * Number(l.prixUnitaireHt) * Number(l.tauxTva) / 100), 0);
    
    return {
      montantHt: Math.round(montantHt * 100) / 100,
      montantTva: Math.round(montantTva * 100) / 100,
      montantTtc: Math.round((montantHt + montantTva) * 100) / 100,
    };
  }

  static async genererNumeroFacture(clinicId: string): Promise<string> {
    const year = new Date().getFullYear();
    const month = String(new Date().getMonth() + 1).padStart(2, '0');
    
    const result = await db.execute(
      sql`SELECT COUNT(*) as count FROM factures 
          WHERE clinic_id = ${clinicId} 
          AND EXTRACT(YEAR FROM date_facture) = ${year}
          AND EXTRACT(MONTH FROM date_facture) = ${new Date().getMonth() + 1}`
    );
    
    const count = Number((result.rows[0] as any).count) + 1;
    return `FAC-${year}${month}-${String(count).padStart(4, '0')}`;
  }

  static async validerClinicOwnership(factureId: number, clinicId: string): Promise<void> {
    const facture = await db.select()
      .from(factures)
      .where(and(eq(factures.id, factureId), eq(factures.clinicId, clinicId)))
      .limit(1);
    
    if (!facture.length) {
      throw new AppError(404, 'Facture introuvable', 'NOT_FOUND');
    }
  }
}
```

**`server/services/stockService.ts`**
```typescript
export class StockService {
  static async verifierDisponibilite(
    stockMedicamentId: number, 
    quantiteRequise: number,
    clinicId: string
  ): Promise<void> {
    const stock = await db.select()
      .from(stockMedicaments)
      .where(and(
        eq(stockMedicaments.id, stockMedicamentId),
        eq(stockMedicaments.clinicId, clinicId)
      ))
      .limit(1);
    
    if (!stock.length) throw new AppError(404, 'Médicament introuvable');
    
    if (stock[0].stockActuel < quantiteRequise) {
      throw new AppError(409, 
        `Stock insuffisant : ${stock[0].stockActuel} disponible, ${quantiteRequise} requis`,
        'STOCK_INSUFFISANT'
      );
    }
    
    if (stock[0].estStupefiant) {
      throw new AppError(400, 
        'Ce médicament est un stupéfiant — utiliser l\'endpoint dédié /api/stupefiants',
        'STUPEFIANT_REQUIRED'
      );
    }
  }

  static async creerMouvement(params: {
    stockMedicamentId: number;
    clinicId: string;
    type: 'entree' | 'sortie' | 'ajustement' | 'perte';
    quantite: number;
    motif: string;
    animalId?: number;
    veterinaireId?: number;
    ordonnanceId?: number;
  }): Promise<void> {
    await db.transaction(async (tx) => {
      // Insérer le mouvement
      await tx.insert(mouvementsStock).values({
        ...params,
        createdAt: new Date(),
      });
      
      // Mettre à jour le stock en atomique
      const delta = params.type === 'entree' ? params.quantite : -params.quantite;
      await tx.update(stockMedicaments)
        .set({ 
          stockActuel: sql`stock_actuel + ${delta}`,
          updatedAt: new Date()
        })
        .where(eq(stockMedicaments.id, params.stockMedicamentId));
    });
  }
}
```

### 5.2 — Routes réduites à leur strict minimum
Les routes ne doivent faire que : valider → appeler le service → retourner la réponse. Zéro logique métier dans les routes.

```typescript
// Exemple AVANT (mauvais) — logique métier dans la route :
router.post('/factures/:id/valider', async (req, res) => {
  // 80 lignes de logique ici...
});

// APRÈS (correct) — route = orchestrateur uniquement :
router.post('/factures/:id/valider', 
  requireAuth, 
  validate(validerFactureSchema),
  asyncHandler(async (req, res) => {
    const factureId = parseInt(req.params.id);
    const clinicId = req.auth.sessionClaims?.clinic_id as string;
    
    await FactureService.validerClinicOwnership(factureId, clinicId);
    const montants = await FactureService.recalculerMontants(factureId);
    const result = await FactureService.valider(factureId, req.body, montants);
    
    res.json({ success: true, data: result });
  })
);
```

```tsx
import { Component, ErrorInfo, ReactNode } from 'react';

interface Props { children: ReactNode; }
interface State { hasError: boolean; error?: Error; }

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false };

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    console.error('[ErrorBoundary]', error, info);
    // TODO: envoyer à un service de monitoring (Sentry)
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="flex flex-col items-center justify-center min-h-screen gap-4">
          <h1 className="text-xl font-bold text-red-600">Une erreur est survenue</h1>
          <p className="text-gray-500">Rechargez la page ou contactez le support.</p>
          <button 
            onClick={() => this.setState({ hasError: false })}
            className="px-4 py-2 bg-blue-600 text-white rounded"
          >
            Réessayer
          </button>
        </div>
      );
    }
    return this.props.children;
  }
}
```
